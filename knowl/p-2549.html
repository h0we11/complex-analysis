\begin{align*} h\,'(z) \amp = f(z) g\,'(z) +f\,'(z) g(z) ;\\ h\,''(z) \amp = f\,''(z)g(z) +2f\,'(z) g\,'(z)+f(z) g\,''(z)\text{.} \end{align*}
