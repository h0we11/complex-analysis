\begin{align*} \lim\limits_{z \to 1+i}\left(\frac{z^2-2i}{z^2-2z+2}\right) \amp =\lim\limits_{z \to 1+i}\frac{(z-1-i)(z+1+i)}{(z-1-i) (z-1+i)}\\ \amp =\lim\limits_{z \to 1+i}\left(\frac{z+1+i}{z-1+i}\right)\\ \amp =\frac{(1+i) +1+i}{(1+i) -1+i}\\ \amp =\frac{2+2i}{2i}\\ \amp =1-i \end{align*}
