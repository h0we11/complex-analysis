\begin{align*} e^{i\theta} \amp = \exp(i\theta )\\ \amp = \sum_{n=0}^{\infty}\frac{1}{n!}(i\theta)^n\\ \amp = \sum_{n=0}^{\infty}\left[\frac{1}{(2n)!} (i\theta)^{2n}+\frac{1}{(2n+1)!}(i\theta)^{2n+1}\right] {\text{ (separating odd and even exponents) } }\\ \amp = \sum_{n=0}^{\infty}\left[\frac{1}{(2n)!}(i^2)^n\theta^{2n} + \frac{1}{(2n+1)!}i(i^2 )^n\theta^{2n+1}\right]\\ \amp = \sum_{n=0}^{\infty}(-1)^n \frac{\theta^{2n}}{(2n)!} + i\sum_{n=0}^{\infty}(-1)^n\frac{\theta^{2n+1}}{(2n+1)!}\\ \amp = \cos \theta +i\sin \theta {\text{ (by the series representations for the real-valued sine and cosine) } } \end{align*}
