\begin{equation*} \frac{x^2}{\cosh^2 2}+\frac{y^2}{\sinh^2 2}=1 \end{equation*}

\begin{align*} T(x,y) \amp = 80 \text{ for } (x,y) \text{ on the ellipse, }\\ T(x,0) \amp = 40 \text{ for } -1 \lt x \lt 1,\\ \frac{\partial T}{\partial n} \amp = T_y(x,0) = 0 \text{ when } 1 \lt |x| \lt \cosh 2\text{.} \end{align*}
